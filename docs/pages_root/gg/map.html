<script>
	async function init(){

		let mapData = await d3.json("./data/nycJSON.json")
    let cloroData = await d3.json("./data/cloroJSON.json")

		// unify geoJSON and style data
		cloroData = cloroData.map((item, i) => {
			let featureId = item.id;
			let feature = mapData.features.filter(d => d.id == featureId)[0];
			return {...item, feature:feature};
		});

		const aes = {
			fill:{factor:'val'}
		}

		let g = new H.gg(cloroData, aes);
			g.geom_map();
			g.scale_fill_brewer({palette:'PuBuGn', direction:-1})
			g.theme_void();
			g.print('#my_view');

	}

	init()

</script>
