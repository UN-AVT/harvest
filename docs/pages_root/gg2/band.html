Band Axes

This is a simple chart using a band axis.



Steps:

  • The Html part of the code just creates a div that will be modified by d3 later on.

  • The first part of the javascript code set a svg area. It specify the chart size and its margin. Read more.

<script>
let smData = [{x:'a', y:2}, {x:'b', y:4}, {x:'c', y:6}, {x:'d', y:4}, {x:'e', y:2}];

let cfg = {
  element:{
    renderer:{name:'svg'},
    sizing:{
      width:400,
      height:400,
    }
  },
  accessors:{
    x:{domain:['a','b','c','d','e'], type:'band', range:'width'},
    y:{domain:[0,6]},
    color:{field:'x', domain:['a','b','c','d','e'], type:'ordinal', range:['#bada55', '#7fe5f0', '#ff0000', '#ff80ed', '#407294']}
  },
  data:[{content:smData}],
  draw:[
    {name:'axis', scale:'x', ticks:{stroke:{width:2, color:'orange'}}, dbar:{stroke:{color:'white'}}},
    {name:'axis', scale:'y', orientation:'left'},
    {name:'rectangle', color:'color', x:'x', y:'y'}
  ]
}

let c0 = new H.Visualization(cfg);

</script>
